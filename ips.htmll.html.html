  document.getElementById("detailPage").classList.add("active");
}

function tampilSoal() {
  const soal = document.getElementById("soal");
  const pilihan = document.getElementById("pilihan");
  const hasil = document.getElementById("hasilQuiz");

  sudahJawab = false;
  hasil.innerHTML = "";

  soal.innerHTML = "❓ " + quiz[indexSoal].soal;
  pilihan.innerHTML = "";

  quiz[indexSoal].pilihan.forEach((p, i) => {
    pilihan.innerHTML += `
      <button onclick="cekJawaban(${i})">${p}</button>
    `;
  });
}

function cekJawaban(pilih) {
  if (sudahJawab) return;

  sudahJawab = true;

  const hasil = document.getElementById("hasilQuiz");

  if (pilih === quiz[indexSoal].jawaban) {
    hasil.innerHTML = "🎉 Benar!";
    hasil.style.color = "green";
    score++;
  } else {
    hasil.innerHTML = "❌ Salah!";
    hasil.style.color = "red";
  }
}

function nextSoal() {
  if (!sudahJawab) {
    alert("Jawab dulu soal ini!");
    return;
  }

  indexSoal++;

  if (indexSoal >= quiz.length) {
    document.getElementById("soal").innerHTML = "🏆 Quiz selesai!";
    document.getElementById("pilihan").innerHTML = "";

    let nilai = Math.round((score / quiz.length) * 100);

    document.getElementById("hasilQuiz").innerHTML =
      `Nilai kamu: <b>${nilai}</b> 🎯<br>Benar: ${score} dari ${quiz.length}`;

    return;
  }

  tampilSoal();
}
